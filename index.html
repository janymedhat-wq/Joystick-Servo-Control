U8G2_SSD1306_128X64_NONAME_F_HW_I2C u8g2(U8G2_R0, /* reset=*/ U8X8_PIN_NONE);

// Joystick Analog Pins
const int JOY_X_PIN = A0; 
const int JOY_Y_PIN = A1; 
const int JOY_SW_PIN = 2; // Joystick button for START/PAUSE

// Game state variables
int pacmanX = 64; // Starting X coordinate (center of 128)
int pacmanY = 32; // Starting Y coordinate (center of 64)
const int MOVE_SPEED = 1;
const int DEADZONE = 500; // Analog value threshold (out of 1023)

void setup() {
  // Initialize the OLED display
  u8g2.begin();

  // Setup joystick digital button
  pinMode(JOY_SW_PIN, INPUT_PULLUP);
}

void loop() {
  // 1. Read Joystick Input (0 - 1023)
  int xValue = analogRead(JOY_X_PIN);
  int yValue = analogRead(JOY_Y_PIN);

  // 2. Determine Movement Direction based on Deadzone
  int dx = 0;
  int dy = 0;

  if (xValue < 1023 / 2 - DEADZONE / 2) {
    dx = -MOVE_SPEED; // Move Left
  } else if (xValue > 1023 / 2 + DEADZONE / 2) {
    dx = MOVE_SPEED;  // Move Right
  }

  if (yValue < 1023 / 2 - DEADZONE / 2) {
    dy = MOVE_SPEED;  // Move Up (Note: Y=0 is usually top of screen)
  } else if (yValue > 1023 / 2 + DEADZONE / 2) {
    dy = -MOVE_SPEED; // Move Down
  }

  // 3. Update Pac-Man Position
  pacmanX += dx;
  pacmanY += dy;

  // 4. Boundary Check (Keep Pac-Man on the 128x64 screen)
  pacmanX = constrain(pacmanX, 0, 127);
  pacmanY = constrain(pacmanY, 0, 63);

  // 5. Drawing routine (U8G2 Page Loop)
  u8g2.firstPage();
  do {
    // Draw the static maze (Conceptual - replace with actual maze drawing logic)
    u8g2.drawFrame(0, 0, 128, 64);
    
    // Draw Pac-Man (a simple filled circle)
    u8g2.drawDisc(pacmanX, pacmanY, 4);

    // Placeholder for drawing dots, ghosts, and score
    // u8g2.setFont(u8g2_font_helvR08_tf);
    // u8g2.drawStr(1, 63, "SCORE: 0000");

  } while (u8g2.nextPage());

  // Small delay to control the frame rate and stabilize input
  delay(20); 
}

CIRCUIT SCHEMATIC

THE SETUP REQUIRES CONNECTIONS FOR THE JOYSTICK MODULE (ANALOG AND DIGITAL INPUTS) AND AN I2C OLED DISPLAY (SERIAL COMMUNICATION).

  • JOYSTICK: $V_{R_X} \rightarrow$ Analog A0. $V_{R_Y} \rightarrow$ Analog A1. SW $\rightarrow$ Digital Pin 2. VCC $\rightarrow$ 5V. GND $\rightarrow$ GND.
  • I2C OLED DISPLAY: SDA $\rightarrow$ Analog A4 (UNO). SCL $\rightarrow$ Analog A5 (UNO). VCC $\rightarrow$ 5V. GND $\rightarrow$ GND.

KEY LEARNING OUTCOMES

THE FOLLOWING CONCEPTS WERE MASTERED DURING THE DEVELOPMENT AND PROTOTYPING OF THIS MINIMALIST EMBEDDED GAME SYSTEM.

2D INPUT AND DEADZONES

LEARNING TO PROCESS TWO ANALOG INPUTS ($V_{R_X}$ AND $V_{R_Y}$) SIMULTANEOUSLY AND IMPLEMENTING A SOFTWARE DEADZONE TO PREVENT PAC-MAN DRIFTING WHEN THE JOYSTICK IS CENTERED.

STATE MANAGEMENT & GAME LOOP

MASTERING THE CORE GAME LOOP STRUCTURE: INPUT $\rightarrow$ UPDATE STATE $\rightarrow$ RENDER OUTPUT, AND MANAGING BASIC GAME ENTITIES LIKE PLAYER POSITION AND BOUNDARIES.

MONOCHROME DISPLAY DRIVING

UTILIZING LIBRARIES (E.G., U8G2) TO INTERFACE WITH SERIAL DISPLAYS (I2C/SPI), UNDERSTANDING THE CONCEPTS OF PAGE BUFFERING AND FRAME-BY-FRAME RENDERING.

CODE OPTIMIZATION & CONSTRAINTS

OPTIMIZING CODE FOR LIMITED ARDUINO MEMORY (SRAM/FLASH) AND USING THE `constrain()` FUNCTION TO ENSURE GAME ENTITIES REMAIN WITHIN THE DEFINED GAME BORDERS.
